44/* ===============================================================================================
45 Frechet
46
47 Notes:
48 =====
49 curve1 and curve2 are two sets of points that define 2 polygonal curves
50 The points along the curves are expected to be in the order as they appear.
51
52 Returned is the discrete Frechet distance, i.e. the coupling
53 measure, which is zero when curve1 equals curve2 and grows positively as the curves
54 become more dissimilar.
55
56 The L2 norm is used to compute the distances between points in curve1 and curve2.
57
58 Explanation:
59 ===========
60 The Frechet distance is a measure of similarity between to curves.
61 It is defined as the minimum cord-length sufficient to join a point
62 traveling forward along curve1 and one traveling forward along curve2, although the
63 rate of travel for either point may not necessarily be uniform.
64
65 The Frechet distance, is not in general computable for any given
66 continuous curves . However, the discrete Frechet Distance, also called
67 the coupling measure, is a metric that acts on the endpoints of
68 curves represented as polygonal chains. The magnitude of the coupling
69 measure is bounded by the continuous Frechet distance plus the length of the
70 longest segment in either curve1 or curve2.
71
72 This function implements the algorithm to calculate discrete Frechet
73 distance outlined table 1 in:
74 T. Eiter and H. Mannila. Computing discrete Frechet distance. Technical
75 Report 94/64, Christian Doppler Laboratory, Vienna University of
76 Technology, 1994.
77
78 It loosely follows a matlab implementation:
79 Zachary Danziger (2021). Discrete Frechet Distance
80 (https://www.mathworks.com/matlabcentral/fileexchange/31922-discrete-frechet-distance),
81 MATLAB Central File Exchange. Retrieved October 15, 2021.
82
83 =============================================================================================== */

85 double Frechet::dFD(Polygon& curve1, Polygon& curve2)
86 {
87
88 // Number of points in the two curves
89
90 int n1 = curve1.vertices.size();
91 int n2 = curve2.vertices.size();
92 int n1n2 = n1*n2;
93
94 // Initialize a matrix CA of size n1 x n2, with all -1
95
96 double *CA = new double[n1*n2];
97 for(int i = 0; i < n1n2; i++) CA[i] = -1;
98
99 // The discrete Frechet distance is the coupling for the last points in curve1 and curve2
100 double d = coupling(curve1, curve2, n1-1, n2-1, CA);
101
102 return d;
103 }

105/* ===============================================================================================
106 Coupling search function
107 Note: this function is recursive
108 =============================================================================================== */

110 double Frechet::coupling(Polygon& curve1, Polygon& curve2, int i, int j, double *CA)
111 {
112 int n1 = curve1.vertices.size();
113 double val;
114
115 if(CA[i+j*n1] > -1 ) return CA[i+j*n1];
116 if (i==0 && j == 0) {
117 CA[i+j*n1] = (curve1.vertices[0].position -curve2.vertices[0].position).norm();
118 } else if (i>0 && j == 0) {
119 CA[i+j*n1] = std::max( coupling(curve1, curve2, i-1, 0, CA), (curve1.vertices[i].position -curve2.vertices[0].position).norm());
120 } else if (i==0 && j > 0) {
121 CA[i+j*n1] = std::max( coupling(curve1, curve2, 0, j-1, CA), (curve1.vertices[0].position -curve2.vertices[j].position).norm());
122 } else if (i>0 && j > 0) {
123 val = std::min( coupling(curve1, curve2, i-1, j, CA), coupling(curve1, curve2, i-1, j-1, CA));
124 val = std::min( val, coupling(curve1, curve2, i, j-1, CA));
125 CA[i+j*n1] = std::max(val, (curve1.vertices[i].position -curve2.vertices[j].position).norm());
126 } else {
127 CA[i+j*n1] = DBL_MAX;
128 }
129 return CA[i+j*n1];
130 }
