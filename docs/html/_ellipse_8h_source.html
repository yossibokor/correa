113/* ===============================================================================================
114 MinVolEllipse
115
116 Finds the minimum volume enclosing ellipsoid of a set of data
117 points stored in a matrix P. The following optimization problem is solved:
118
119 minimize log(det(A))
120 subject to (P_i - c)' * A * (P_i - c) <= 1
121
122 in variables A and c, where P_i is the i-th column of the matrix P.
123 The solver is based on Khachiyan Algorithm, and the final solution
124 is different from the optimal value by the pre-specified amount of 'tol'.
125
126 inputs:
127 ---------
128 P : (d x N) dimensional matrix containing N points in R^d.
129 tol : error in the solution with respect to the optimal value.
130
131 outputs:
132 ---------
133 A : matrix representing ellipse (in fact, inv(A))
134 c : center of ellipse
135
136 Reference:
137 ----------
138 Code based on the Matlab package MinVolEllipse:
139
140 Nima Moshtagh Minimum Volume Enclosing Ellipsoid
141 https://www.mathworks.com/matlabcentral/fileexchange/9542-minimum-volume-enclosing-ellipsoid,
142 MATLAB Central File Exchange.
143
144 and referencing paper:
145
146 Nima Moshtagh, "Minimum volume enclosing ellipsoid"
147 http://citeseerx.ist.psu.edu/viewdoc/summary?doi=10.1.1.116.7691
148
149 =============================================================================================== */
150
151 void Ellipse::MinVolEllipse(double *Points, int Np, int d, double *A, double *c, double tol)
152 {
153 // Build convex hull of the polygon defined by P
154
155 std::vector<Vector2D> CH;
156 std::vector<Vector2D> Poly;
157
158 for(int i = 0; i < Np; i++) {
159 Vector2D v(Points[2*i], Points[2*i+1]);
160 Poly.push_back(v);
161 }
162 convexhull.ConvexPoly(Poly, CH);
163
164 int N = CH.size()-1;
165 double *P = new double[2*N];
166 for(int i = 0; i < N; i++) {
167 P[2*i + 0] = CH[i][0];
168 P[2*i + 1] = CH[i][1];
169 }
170
171 double *Q = new double[(d+1)*N];
172 memset(Q, 0.0, (d+1)*N*sizeof(double));
173
174 for (int j = 0; j < N; j++) {
175 for(int dim = 0; dim < d; dim++) {
176 Q[dim+(d+1)*j] = P[dim+d*j];
177 }
178 Q[d+(d+1)*j] = 1.0;
179 }
180
181 /* =========================================================================================
182 Initialisation
183 ========================================================================================= */
184
185 double err = 2*tol;
186
187 double *u = new double[N];
188 double *new_u = new double[N];
189 for(int i = 0; i < N; i++) u[i] = 1./N;
190
191 double *X = new double[(d+1)*(d+1)];
192 int *IPIV = new int[d+1];
193 int lwork = 100*(d+1);
194 double *WORK = new double[lwork];
195 double *v = new double[d+1];
196 double *w = new double[d+1];
197 double *M = new double[N];
198
199 int dp1 = d+1;
200 int dd = d*d;
201 int info;
202 int inc = 1; int one = 1;
203 char NoTrans = 'N'; char Trans = 'T';
204 double alpha, beta;
205 double dMax, step, sum;
206 int idx;
207
208 /* =========================================================================================
209 Khachiyan Algorithm
210 ========================================================================================= */
211
212 while (err > tol) {
213
214 // X = Q * diag(u) * Q'; is a (d+1)x(d+1) matrix
215
216 for(int i = 0; i < d+1; i++) {
217 for(int j = 0; j < d+1; j++) {
218 sum = 0.0;
219 for(int k = 0; k < N; k++) {
220 sum += Q[i+k*(d+1)]*u[k]*Q[j+k*(d+1)];
221 }
222 X[i+(d+1)*j] = sum;
223 }
224 }
225
226 // M = diag(Q' * inv(X) * Q); % M the diagonal vector of an NxN matrix
227
228 dgetrf_(&dp1, &dp1, X, &dp1, IPIV, &info);
229 dgetri_(&dp1, X, &dp1, IPIV, WORK, &lwork, &info);
230
231 alpha = 1.0; beta = 0;
232 for(int i = 0; i < N; i++) {
233 dcopy_(&dp1, &Q[i*dp1], &inc, v, &inc);
234 dgemv_(&NoTrans, &dp1, &dp1, &alpha, X, &dp1, v, &inc, &beta, w, &inc);
235 M[i] = ddot_(&dp1, v, &inc, w, &inc);
236 }
237
238 dMax = M[0]; idx = 0;
239 for(int i = 1; i < N; i++) {
240 if(M[i] > dMax) {
241 dMax = M[i];
242 idx = i;
243 }
244 }
245
246 // Compute U_new
247
248 step = (dMax -d -1)/(dp1*(dMax-1));
249 alpha = 1.0 - step;
250 memset(new_u, 0.0, N*sizeof(double));
251 daxpy_(&N, &alpha, u, &inc, new_u, &inc);
252 new_u[idx] += step;
253
254 // Check error
255
256 alpha = -1.0;
257 daxpy_(&N, &alpha, new_u, &inc, u, &inc);
258 err = dnrm2_(&N, u, &inc);
259
260 // Update U
261
262 dcopy_(&N, new_u, &inc, u, &inc);
263 }
264
265 /* =========================================================================================
266 Find the ellipse equation
267 (x-c)' * A * (x-c) = 1
268 i.e. compute a dxd matrix 'A' and a d dimensional vector 'c' as the center
269 of the ellipse:
270
271 A = (1/d) * inv(P * diag(u) * P' - (P * u)*(P*u)' )
272 c = P*u
273
274 In fact, we only need inv(A) = d * ( P * diag(u) * P' - (P * u)*(P*u)' )
275 ========================================================================================= */
276
277 // Compute c = P*U
278 alpha = 1.0; beta = 0.0;
279 dgemv_(&NoTrans, &d, &N, &alpha, P, &d, u, &inc, &beta, c, &inc);
280
281 // Compute inv(A) = d * ( P * diag(u) * P' - (P * u)*(P*u)' )
282 for(int i = 0; i < d; i++) {
283 for(int j = 0; j < d; j++) {
284 sum = 0.0;
285 for(int k = 0; k < N; k++) {
286 sum += P[i+k*d]*u[k]*P[j+k*d];
287 }
288 A[i+j*d] = sum;
289 }
290 }
291
292 alpha = -1.0; beta = 1.0;
293 dgemm_(&NoTrans, &Trans, &d, &d, &one, &alpha, c, &d, c, &d, &beta, A, &d);
294
295 alpha = d;
296 dscal_(&dd, &alpha, A, &inc);
297
298 // Clean up
299 delete [] P; delete [] Q; delete [] v; delete [] w; delete [] WORK; delete [] IPIV;
300
301 }

303/* ===============================================================================================
304 Ellipse2D
305
306 Given the matrix representation A of 2D ellipse and its center c
307 (x-c)' * A * (x-c) = 1
308 compute its major a and minor axis b, as well as a discrete representation
309 of the ellipse over 100 points
310
311 inputs:
312 ---------
313 A : (d x d) dimensional matrix (in fact, A^{-1}
314 c : center of ellipse
315
316 outputs:
317 ---------
318 a : size along major axis of the ellipse
319 b : size along minor axis of the ellipse
320 ellipse: discrete representation of the ellipse over 100 points
321
322 =============================================================================================== */
323
324 void Ellipse::Ellipse2D(double *A, double *c, double *a, double *b, double *ellipse)
325 {
326
327 // Diagonalize matrix inv(A)
328 int d = 2;
329 double *w = new double[d];
330
331 int lwork = 2*d*d+6*d+1; double *WORK = new double[lwork];
332 int liwork = 5*d+3; int *IWORK = new int[liwork];
333
334 char JOBZ='V'; char UPLO='U';
335 int info;
336 dsyevd_(&JOBZ, &UPLO, &d, A, &d, w, WORK, &lwork, IWORK, &liwork, &info);
337
338 // Major and Minor axis are the square roots of the eigenvalues
339 *a = std::sqrt(w[1]);
340 *b = std::sqrt(w[0]);
341
342 // Orientation of the ellipse defined from the eigenvector associated with major axis
343 int inc = 1;
344 dcopy_(&d, &A[d], &inc, w, &inc);
345
346 double ang = std::atan2(w[1],w[0]);
347
348 // Compute discrete represention of ellipse over 100 points
349 int ntime = 100;
350 double *t = new double[ntime];
351 double step = 2*M_PI/(ntime-1);
352 for(int i = 0; i < ntime; i++) t[i] = -M_PI + i*step;
353
354 double x1, x2;
355 for(int i = 0; i < ntime; i++) {
356 x1 = (*a)*std::cos(t[i]);
357 x2 = (*b)*std::sin(t[i]);
358 ellipse[2*i] = std::cos(ang)*x1 - std::sin(ang)*x2 + c[0];
359 ellipse[2*i+1] = std::sin(ang)*x1 + std::cos(ang)*x2 + c[1];
360 }
361
362 // Clean up
363 delete [] WORK; delete [] IWORK; delete [] t;
364
365 }

367/* ===============================================================================================
368 MaxVolEllipse
369
370 Finds the maximum volume inclosing ellipsoid of a convex polygon.
371 As the input is a general polygon, first computes the convex hull of that polygon
372 The following optimization problem is solved:
373
374 Find the maximum volume ellipsoid
375 {v: v = x + Es, ||s|| <= 1}
376 inscribing a full-dimensional polytope
377 {v: Av <= b}
378
379 inputs:
380 ---------
381 P : (d x N) dimensional matrix containing N points in R^d.
382 tol : error in the solution with respect to the optimal value.
383
384 outputs:
385 ---------
386 E : matrix representing ellipse (in fact, E*E')
387 c : center of ellipse
388
389 Reference:
390 ----------
391 Implementation based on:
392
393 Yin Zhang and Liyan Gao. On Numerical Solution of the Maximum Volume Ellipsoid Problem.
394 SIAM Journal on Optimization, Vol.14, No.1, pp. 53-76, 2003.
395
396 and corresponding MATLAB code available at:
397 https://www.caam.rice.edu/~zhang/mve/index.html
398
399 =============================================================================================== */
400
401 void Ellipse::MaxVolEllipse(double *P, int N, int d, double *E2, double *x, double tol)
402 {
403
404 // parameters
405 int maxiter = 50;
406 double minmu = 1.e-8;
407 double tau0 = 0.75;
408 int iprint = 0;
409
410 // Build convex hull of the polygon defined by P
411
412 std::vector<Vector2D> CH;
413 std::vector<Vector2D> Poly;
414
415 for(int i = 0; i < N; i++) {
416 Vector2D v(P[2*i], P[2*i+1]);
417 Poly.push_back(v);
418 }
419 convexhull.ConvexPoly(Poly, CH);
420 int nCH = CH.size();
421
422 // Center of gravity of convex hull
423
424 double x0[2];
425 x0[0] = 0; x0[1] = 0;
426 for(int i = 0; i < nCH-1; i++) {
427 x0[0] += CH[i][0];
428 x0[1] += CH[i][1];
429 }
430 x0[0] /= (nCH-1); x0[1] /= (nCH-1);
431
432 // Generate affine constraints A and b
433
434 int n = nCH-1; int nn=n*n;
435 double *A = new double[n*d];
436 double *b = new double[n];
437 double *bmAx0 = new double[n];
438 double r, s, t, u, val;
439 for(int i = 0; i < n; i++) {
440 r = CH[i+1][0]-CH[i][0]; s = CH[i+1][1]-CH[i][1];
441 val = std::sqrt(r*r+s*s);
442 t = r; r = -s/val; s = t/val;
443 t = CH[i+1][0]+CH[i][0];
444 u = CH[i+1][1]+CH[i][1];
445 A[i] = r; A[i+n] = s;
446 b[i] = 0.5*(r*t+s*u);
447 val = r*x0[0]+s*x0[1] -b[i];
448 if(val > 0) {
449 A[i] = -A[i]; A[i+n]=-A[i+n]; b[i] = -b[i];
450 }
451 }
452
453 int inc = 1;
454 double alpha = -1; double beta = 1;
455 char NoTrans = 'N'; char Trans = 'T';
456
457 double bnrm = dnrm2_(&n, b, &inc);
458 dcopy_(&n, b, &inc, bmAx0, &inc);
459 dgemv_(&NoTrans, &n, &d, &alpha, A, &n, x0, &inc, &beta, bmAx0, &inc);
460
461 // Check that center of gravity, Xmean, is inside the convex polygon
462 // Should never be a problem...
463
464 for(int i = 0; i < n; i++) {
465 if(bmAx0[i] <= 0) {
466 std::cout << "Problem in MaxVolEllipse!" << std::endl;
467 std::cout << "Center of gravity not in convex hull...." << std::endl;
468 exit(1);
469 }
470 }
471
472 for(int i = 0; i < n; i++) {
473 for(int k = 0; k < d; k++) {
474 A[i+k*n] /= bmAx0[i];
475 }
476 b[i] = 1.0;
477 }
478
479 double *y = new double[n]; double *z = new double[n];
480 double *dx = new double[d]; double *dy = new double[n]; double *dz = new double[n];
481 double *yz = new double[n]; double *yh = new double[n]; double *bmAx = new double[n];
482
483 memset(x, 0, d*sizeof(double));
484 for (int i = 0; i < n; i++) {
485 y[i] = 1.0;
486 bmAx[i] = 1.0;
487 }
488
489 double res = 1;
490 double sum, gap, rmu;
491
492 double *Ei = new double[d*d];
493 double *W = new double[n*d];
494 double *Q = new double[n*n]; double *G = new double[n*n];
495 double *h = new double[n]; double *Adx = new double[n];
496 double *R1 = new double[d]; double *R2 = new double[n]; double *R3 = new double[n];
497 int *IPIV = new int[n];
498
499 int info;
500 double r1, r2, r3, objval;
501 double ax, ay, az, tau, astep;
502
503 if(iprint==1) {
504 std::cout << "=========================================================" << std::endl;
505 std::cout << " Residuals: Primal Dual Duality log(detE) " << std::endl;
506 std::cout << "=========================================================" << std::endl;
507 }
508
509 // Start iterations
510 for(int iter = 0; iter < maxiter; iter++) {
511
512 if(iter > 0) {
513 alpha = -astep;
514 daxpy_(&n, &alpha, Adx, &inc, bmAx, &inc);
515 }
516
517 for(int i = 0; i < d; i++) {
518 for(int j = 0; j < d; j++) {
519 sum = 0.;
520 for(int k = 0; k < n; k++) {
521 sum+= A[k+i*n]*y[k]*A[k+j*n];
522 }
523 Ei[i+d*j] = sum;
524 }
525 }
526 matInv2(Ei, E2);
527
528 alpha = 1.0; beta = 0.0;
529 dgemm_(&NoTrans, &NoTrans, &n, &d, &d, &alpha, A, &n, E2, &d, &beta, W, &n);
530 dgemm_(&NoTrans, &Trans, &n, &n, &d, &alpha, W, &n, A, &n, &beta, Q, &n);
531 for(int i = 0; i < n; i++) {
532 h[i] = std::sqrt(Q[i+n*i]);
533 }
534
535 if(iter == 0) {
536 t = bmAx[0]/h[0];
537 for(int i = 1; i < n; i++) t = std::min(t, bmAx[i]/h[i]);
538 for(int i = 0; i < n; i++) {
539 y[i] = y[i]/(t*t);
540 h[i] = t*h[i];
541 z[i] = std::max(0.1, bmAx[i]-h[i]);
542 }
543 alpha = t*t;
544 dscal_(&nn, &alpha, Q, &inc);
545 }
546
547 sum = 0.0;
548 for(int i = 0; i < n; i++) {
549 yz[i] = y[i]*z[i];
550 yh[i] = y[i]*h[i];
551 sum += yz[i];
552 }
553 gap = sum/n;
554 rmu = std::min(0.5, gap)*gap;
555 rmu = std::max(rmu, minmu);
556
557 alpha = -1.0; beta = 0.0;
558 dgemv_(&Trans, &n, &d, &alpha, A, &n, yh, &inc, &beta, R1, &inc);
559 r1 = std::max(std::abs(R1[0]), std::abs(R1[1]));
560 r2 = 0; r3 = 0;
561 for(int i = 0; i < n; i++) {
562 R2[i] = bmAx[i] - h[i] - z[i];
563 R3[i] = rmu - yz[i];
564 r2 = std::max(r2, std::abs(R2[i]));
565 r3 = std::max(r3, std::abs(R3[i]));
566 }
567 res = std::max(r1, std::max(r2, r3));
568 val = E2[0]*E2[3]-E2[1]*E2[2];
569 objval = std::log(val)/2;
570
571 if(iprint==1) {
572 std::cout << "iter = " << iter << " " << r2 << " " << r1 << " " << r3 << " " << objval << std::endl;
573 }
574
575 if( res < tol*(1+bnrm) && rmu <= minmu) {
576 for(int i = 0; i < d; i++) x[i] += x0[i];
577 break;
578 }
579
580 for(int i = 0; i < n; i++) {
581 for(int j = 0; j < n; j++) {
582 G[i+n*j] = y[i]*y[j]*Q[i+j*n]*Q[j+i*n];
583 }
584 G[i+i*n] += std::max(1.e-12, 2*yh[i]*z[i]);
585 for(int j = 0; j < d; j++) {
586 W[i+j*n] = y[i]*(h[i]+z[i])*A[i+j*n];
587 }
588 }
589
590 dgetrf_(&n, &n, G, &n, IPIV, &info);
591 dgetrs_(&NoTrans, &n, &d, G, &n, IPIV, W, &n, &info);
592
593 for(int i = 0; i < n; i++) {
594 for(int j = 0; j < d; j++) {
595 W[i+j*n] = 2*yh[i]*W[i+j*n] - y[i]*A[i+j*n];
596 }
597 }
598
599 for(int i = 0; i < n; i++) {
600 R3[i] = R3[i]/y[i];
601 R2[i] = R2[i] - R3[i];
602 }
603
604 alpha = 1.0; beta = 0.0;
605 dgemm_(&Trans, &NoTrans, &d, &d, &n, &alpha, W, &n, A, &n, &beta, Ei, &d);
606 dgemv_(&Trans, &n, &d, &alpha, W, &n, R2, &inc, &beta, h, &inc);
607 daxpy_(&d, &alpha, R1, &inc, h, &inc);
608 matInv2(Ei, E2);
609
610 dgemv_(&NoTrans, &d, &d, &alpha, E2, &d, h, &inc, &beta, dx, &inc);
611
612 dgemv_(&NoTrans, &n, &d, &alpha, A, &n, dx, &inc, &beta, Adx, &inc);
613 for(int i = 0; i < n; i++) {
614 h[i] = 2*yh[i]*(Adx[i]-R2[i]);
615 }
616 dgetrs_(&NoTrans, &n, &inc, G, &n, IPIV, h, &n, &info);
617 for(int i = 0; i < n; i++) {
618 dy[i] = y[i]*h[i];
619 dz[i] = R3[i] - z[i]*h[i];
620 }
621
622 ax = Adx[0]/bmAx[0]; ay = h[0]; az = dz[0]/z[0];
623 for(int i = 1; i < n; i++) {
624 ax = std::min(ax, Adx[i]/bmAx[i]); ay = std::min(ay, h[i]); az = std::min(az, dz[i]/z[i]);
625 }
626 ax = std::min(-0.5, ax); ay = std::min(-0.5, ay); az = std::min(-0.5, az);
627 ax = -1/ax; ay = -1/ay; az = -1/az;
628 tau = std::max(tau0, 1-res);
629 astep = std::min(ax, std::min(ay, az)); astep = tau*std::min(1., astep);
630
631 daxpy_(&d, &astep, dx, &inc, x, &inc);
632 daxpy_(&n, &astep, dy, &inc, y, &inc);
633 daxpy_(&n, &astep, dz, &inc, z, &inc);
634 }
635
636
637 if(iprint==1) {
638 std::cout << "=========================================================" << std::endl;
639 std::cout << std::endl;
640 }
641

656/* ===============================================================================================
657 Computes the least square fitting helix
658
659 inputs:
660 ---------
661 P : (d x N) dimensional matrix containing N points in R^d.
662
663 outputs:
664 ---------
665 a : major axis of the ellipse
666 b : minor axis of the ellipse
667
668 Implementation based on:
669
670 A. W. Fitzgibbon, M. Pilu and R. B. Fisher, "Direct least squares fitting of ellipses,"
671 Proceedings of 13th International Conference on Pattern Recognition, 1996, pp. 253-257 vol.1,
672 doi: 10.1109/ICPR.1996.546029.
673
674 =============================================================================================== */
675
676 void Ellipse::lsqEllipse(double *P, int N, int d, double *a, double *b)
677 {
678
679 double *Dmat = new double[6*N];
680 double *Smat = new double[6*6];
681 double *Cmat = new double[6*6];
682 double *VR = new double[6*6];
683 double *VL = new double[6*6];
684 int *IPIV = new int[6];
685 int lwork = 200;
686 double *WORK = new double[lwork];
687
688 // Build design matrix (size 6xN)
689 for(int i = 0; i < N; i++) {
690 Dmat[i + 0*N] = P[2*i]*P[2*i];
691 Dmat[i + 1*N] = P[2*i]*P[2*i+1];
692 Dmat[i + 2*N] = P[2*i+1]*P[2*i+1];
693 Dmat[i + 3*N] = P[2*i];
694 Dmat[i + 4*N] = P[2*i+1];
695 Dmat[i + 5*N] = 1.;
696 }
697
698 // Build scatter matrix Smat = Dmat^T Dmat (size 6x6)
699 int info;
700 int six = 6;
701 char NoTrans = 'N'; char Trans = 'T';
702 double alpha, beta;
703 alpha = 1.0; beta = 0.;
704 dgemm_(&Trans, &NoTrans, &six, &six, &N, &alpha, Dmat, &N, Dmat, &N, &beta, Smat, &six);
705
706 // Build constraint matrix Cmat (size 6x6)
707 memset(Cmat, 0, 6*6*sizeof(double));
708 Cmat[12] = 2; Cmat[7] = -1; Cmat[2]=2;
709
710 // Build eigen problem: matrix stored E = inv(Smat) * C stored in Dmat
711 dgetrf_(&six, &six, Smat, &six, IPIV, &info);
712 dgetri_(&six, Smat, &six, IPIV, WORK, &lwork, &info);
713
714 alpha = 1.0; beta = 0.;
715 dgemm_(&NoTrans, &NoTrans, &six, &six, &six, &alpha, Smat, &six, Cmat, &six, &beta, Dmat, &six);
716
717 // Solve general eigen problem; we only need the right eigenvectors
718 char JOBVL='N'; char JOBVR='V';
719 dgeev_(&JOBVL, &JOBVR, &six, Dmat, &six, Cmat, Smat, VL, &six, VR, &six, WORK, &lwork, &info);
720
721 // Find the positive eigenvalue
722 int idx = -1;
723 for(int i = 0; i < 6; i++) {
724 if(Cmat[i] > 0) {
725 idx = i;
726 break;
727 }
728 }
729
730 // Find coefficient of ellipse: A X^2 + B XY C Y^2 + D X + E Y + F
731 // as the (right) eigenvector corresponding to the positive eigenvalue
732 double A = VR[0+6*idx]; double B = VR[1+6*idx]; double C = VR[2+6*idx];
733 double D = VR[3+6*idx]; double E = VR[4+6*idx]; double F = VR[5+6*idx];
734
735 // Compute major axis (a) and minor axis (b) for the ellipse
736 // see: https://en.wikipedia.org/wiki/Ellipse
737 double den = B*B - 4*A*C;
738 double val1 = (A-C)*(A-C) + B*B;
739 double valp = A+C + std::sqrt(val1);
740 double valn = A+C - std::sqrt(val1);
741 double val2 = A*E*E + C*D*D - B*D*E + (B*B-4*A*C)*F;
742 double nump = std::sqrt(2*val2*valp);
743 double numn = std::sqrt(2*val2*valn);
744
745 val1 = -nump/den;
746 valp = -numn/den;
747
748 *a = std::max(val1,valp);
749 *b = std::min(val1,valp);
750
751 // Clean up
752 delete [] Dmat; delete [] Smat; delete [] Cmat; delete [] VL; delete [] VR; delete [] WORK;
753
754 }
