51 /* ===============================================================================================
52 Replace all instances of a substring by another string
53 =============================================================================================== */
54
55 std::string INOUT::replaceAll(std::string str, const std::string &from, const std::string &to)
56 {
57 size_t start_pos = 0;
58 while ((start_pos = str.find(from, start_pos)) != std::string::npos)
59 {
60 str.replace(start_pos, from.length(), to);
61 start_pos += to.length(); // Handles case where 'to' is a substring of 'from'
62 }
63 return str;
64 }

96/* ===============================================================================================
97 Count # of data points in input file
98 =============================================================================================== */
99
100void INOUT::countPoints(std::ifstream & inFile, int *ndim, int *npoint)
101{
102 std::string line, line0;
103
104 *npoint = 0;
105 int nval=0;
106 double val;
107
108/* ==========================================================================================
109 Go over each line in the file
110 ========================================================================================== */
111
112 while (getline(inFile, line0)) // until reach the end of file
113 {
114/* ==================================================================================
115 Only keep lines that do not contain //
116 ================================================================================== */
117
118 if (line0.substr(0,2) != "//")
119 {
120 *npoint = *npoint + 1;
121
122/* ==========================================================================
123 Count number of records on this line: transform the line into a stringstream
124 and go over the number of record
125 ========================================================================== */
126
127 line = replaceAll(line0, ",", " ");
128 std::istringstream iss(line);
129 nval = 0;
130 while ( iss >> val)
131 {
132 nval = nval + 1;
133 }
134
135 }
136 }
137 *ndim = nval;
138
139}

142 Read points from input file
143 - coordinates x,y (and maybe z)
144 =============================================================================================== */
145
146void INOUT::readPoints(std::ifstream & inFile, int ndim, int npoint, double *coord)
147{
148 std::string line0, line;
149
150 int nat=0;
151 double val;
152
153 while (getline(inFile, line0))
154 {
155 if (line0.substr(0,2) != "//")
156 {
157 line = replaceAll(line0, ",", " ");
158 std::istringstream iss(line);
159 int ncoord = 0;
160 while (iss >> val)
161 {
162 coord[ndim*nat + ncoord] = val;
163 ncoord++;
164 }
165 nat++;
166 }
167 }
168}
